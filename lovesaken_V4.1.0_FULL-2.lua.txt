local function safeSpawn(fn, ...)
    local a = {...}
    task.spawn(function() pcall(function() fn(unpack(a)) end) end)
end

local function processInBatches(items, cb, bs, delay)
    pcall(function()
        bs = bs or 3; delay = delay or 0.15
        if #items == 0 then return end
        local i = 1
        local function go()
            local n = 0
            while i <= #items and n < bs do
                pcall(function() cb(items[i]) end)
                i = i + 1; n = n + 1
            end
            if i <= #items then task.delay(delay, go) end
        end
        go()
    end)
end

local function safePairs(t)
    if type(t) == "table" then return pairs(t) end
    return function() return nil end
end

-- ============================================================
--  FACING CHECK
-- ============================================================
local function isStrictlyFacing(myRoot, targetRoot, killerName)
    local s, r = pcall(function()
        if not facingCheckEnabled then return true end
        if not myRoot or not targetRoot then return false end
        local dir  = (myRoot.Position - targetRoot.Position).Unit
        local dot  = targetRoot.CFrame.LookVector:Dot(dir)
        local head = targetRoot.Parent and targetRoot.Parent:FindFirstChild("Head")
        local hdot = head and head.CFrame.LookVector:Dot(dir) or dot
        local bait = false
        if killerName then
            for _, n in ipairs(BAIT_KILLERS) do if killerName:find(n) then bait = true; break end end
        end
        if bait then
            local side = math.abs((targetRoot.Velocity or Vector3.zero):Dot(targetRoot.CFrame.RightVector))
            if side > 3 then return false end
            return dot > STRICT_FACING_DOT + 0.05 and hdot > STRICT_FACING_DOT - 0.05
        end
        return dot > STRICT_FACING_DOT and hdot > STRICT_FACING_DOT - 0.05
    end)
    return s and r or false
end

local function getNearestKillerModel()
    local s, r = pcall(function()
        local myChar = LocalPlayer.Character; if not myChar then return nil end
        local myRoot = myChar:FindFirstChild("HumanoidRootPart"); if not myRoot then return nil end
        local pf = Workspace:FindFirstChild("Players"); if not pf then return nil end
        local kf = pf:FindFirstChild("Killers"); if not kf then return nil end
        local best, bestD = nil, math.huge
        for _, k in pairs(kf:GetChildren()) do
            local hrp = k:FindFirstChild("HumanoidRootPart")
            if hrp then
                local d = (hrp.Position - myRoot.Position).Magnitude
                if d < bestD then best, bestD = k, d end
            end
        end
        return best
    end)
    return s and r or nil
end

local function beginDragIntoKiller(killerModel)
    if _hitboxDragDebounce then return end
    if not killerModel or not killerModel.Parent then return end
    local char = LocalPlayer.Character; if not char then return end
    local hrp  = char:FindFirstChild("HumanoidRootPart")
    local hum  = char:FindFirstChildOfClass("Humanoid")
    if not hrp or not hum then return end
    local tHRP = killerModel:FindFirstChild("HumanoidRootPart"); if not tHRP then return end
    _hitboxDragDebounce = true
    local oldW, oldJ = hum.WalkSpeed, hum.JumpPower
    hum.WalkSpeed = 0; hum.JumpPower = 0
    local bv = Instance.new("BodyVelocity")
    bv.MaxForce = Vector3.new(1e5, 0, 1e5); bv.Velocity = Vector3.zero; bv.Parent = hrp
    local conn
    conn = RunService.Heartbeat:Connect(function()
        if not _hitboxDragDebounce then
            conn:Disconnect()
            if bv and bv.Parent then bv:Destroy() end
            hum.WalkSpeed = oldW; hum.JumpPower = oldJ
            return
        end
        if not (char and char.Parent) or not (killerModel and killerModel.Parent) then
            _hitboxDragDebounce = false; return
        end
        tHRP = killerModel:FindFirstChild("HumanoidRootPart")
        if not tHRP then _hitboxDragDebounce = false; return end
        local to = tHRP.Position - hrp.Position
        local h2 = Vector3.new(to.X, 0, to.Z)
        bv.Velocity = h2.Magnitude > 0.01 and h2.Unit * Dspeed or Vector3.zero
        if to.Magnitude <= 2.0 then _hitboxDragDebounce = false end
    end)
    task.delay(0.4, function() _hitboxDragDebounce = false end)
end

local function startChargeAim(fallback)
    local sw = tick(); fallback = fallback or 1.2
    local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
    local root = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if hum then hum.AutoRotate = false end
    while tick() - sw < fallback do
        local nk = getNearestKillerModel()
        if nk and root then
            local tHRP = nk:FindFirstChild("HumanoidRootPart")
            if tHRP then
                if rotateDelay > 0 then task.wait(rotateDelay) end
                root.CFrame = CFrame.lookAt(root.Position, tHRP.Position)
            end
        end
        task.wait()
    end
    if hum then hum.AutoRotate = true end
end

    end)
end

local function taphIsTripwire(n) n = n:lower(); return n:find("tripwire") ~= nil end
local function taphIsTripmine(n) n = n:lower(); return n:find("tripmine") ~= nil or n:find("subspace") ~= nil end

local function isPlayerDescendant(obj)
    for _, plr in ipairs(Players:GetPlayers()) do
        local ch = plr.Character
        if ch and obj:IsDescendantOf(ch) then return true end
    end
    return false
end

local function findTrapRoot(obj)
    if obj:IsA("BasePart") then
        return (not isPlayerDescendant(obj)) and obj or nil
    end
    if obj.PrimaryPart and not isPlayerDescendant(obj.PrimaryPart) then return obj.PrimaryPart end
    for _, tag in ipairs({"Handle","Wire","Mine","Plate","Base","Trigger","Part"}) do
        local p = obj:FindFirstChild(tag)
        if p and p:IsA("BasePart") and not isPlayerDescendant(p) then return p end
    end
    for _, d in ipairs(obj:GetDescendants()) do
        if d:IsA("BasePart") and not isPlayerDescendant(d) then return d end
    end
    return nil
end

local function taphApplyESP(obj, cfg)
    pcall(function()
        if not obj or not obj.Parent then return end
        if isPlayerDescendant(obj) then return end
        local root = findTrapRoot(obj); if not root then return end
        if root:FindFirstChild("_TaphESP") then return end

        local tag = Instance.new("BoolValue"); tag.Name = "_TaphESP"; tag.Parent = root

        local hl = Instance.new("Highlight")
        hl.FillColor           = cfg.color
        hl.FillTransparency    = 0.25
        hl.OutlineColor        = cfg.color
        hl.OutlineTransparency = 0.0
        hl.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
        hl.Adornee             = root
        hl.Parent              = root

        local bb = Instance.new("BillboardGui")
        bb.Name        = "_TaphLabel"
        bb.Adornee     = root
        bb.Size        = UDim2.new(0, 120, 0, 20)
        bb.StudsOffset = Vector3.new(0, -0.6, 0)
        bb.AlwaysOnTop = true
        bb.MaxDistance = 80
        bb.Parent      = root

        local lbl = Instance.new("TextLabel")
        lbl.Size                   = UDim2.new(1, 0, 1, 0)
        lbl.BackgroundTransparency = 1
        lbl.Text                   = cfg.label
        lbl.TextColor3             = cfg.color
        lbl.TextStrokeColor3       = Color3.new(0, 0, 0)
        lbl.TextStrokeTransparency = 0.2
        lbl.TextSize               = 11
        lbl.Font                   = Enum.Font.GothamBold
        lbl.Parent = bb

        local cleanConn
        cleanConn = root.AncestryChanged:Connect(function()
            if root.Parent then return end
            pcall(function() cleanConn:Disconnect() end)
            pcall(function() hl:Destroy(); tag:Destroy(); bb:Destroy() end)
            TaphESP.tripwire.tracked[obj] = nil
            TaphESP.tripmine.tracked[obj] = nil
        end)
    end)
end

local function taphRemoveESP(obj)
    pcall(function()
        if not obj then return end
        local function cleanPart(p)
            if not p then return end
            local t = p:FindFirstChild("_TaphESP");    if t  then t:Destroy()  end
            local h = p:FindFirstChildWhichIsA("Highlight"); if h then h:Destroy() end
            local b = p:FindFirstChild("_TaphLabel");  if b  then b:Destroy()  end
        end
        if obj:IsA("BasePart") then
            cleanPart(obj)
        else
            for _, d in ipairs(obj:GetDescendants()) do
                if d:IsA("BasePart") then cleanPart(d) end
            end
        end
    end)
end

local function taphCheckObject(obj)
    pcall(function()
        if not obj or isPlayerDescendant(obj) then return end
        local name = obj.Name
        if taphIsTripwire(name) and TaphESP.tripwire.enabled and not TaphESP.tripwire.tracked[obj] then
            TaphESP.tripwire.tracked[obj] = true
            task.delay(0.15, function() taphApplyESP(obj, TaphESP.tripwire) end)
        end
        if taphIsTripmine(name) and TaphESP.tripmine.enabled and not TaphESP.tripmine.tracked[obj] then
            TaphESP.tripmine.tracked[obj] = true
            task.delay(0.15, function() taphApplyESP(obj, TaphESP.tripmine) end)
        end
    end)
end

local _taphWatched = {}
local function taphWatchFolder(folder)
    pcall(function()
        if not folder or _taphWatched[folder] then return end
        _taphWatched[folder] = true
        folder.ChildAdded:Connect(function(c)
            task.wait(0.1)
            if isPlayerDescendant(c) then return end
            taphCheckObject(c)
            if c:IsA("Model") or c:IsA("Folder") then
                for _, gc in ipairs(c:GetChildren()) do taphCheckObject(gc) end
                c.ChildAdded:Connect(function(gc) task.wait(0.05); taphCheckObject(gc) end)
            end
        end)
        for _, c in ipairs(folder:GetChildren()) do taphCheckObject(c) end
    end)
end

local function setupTaphESPWatchers()
    pcall(function()
        local ingame = Workspace:FindFirstChild("Map") and Workspace.Map:FindFirstChild("Ingame")
        if ingame then
            taphWatchFolder(ingame)
            ingame.ChildAdded:Connect(function(child)
                task.wait(0.1)
                if not isPlayerDescendant(child) then taphWatchFolder(child); taphCheckObject(child) end
            end)
            for _, child in ipairs(ingame:GetChildren()) do taphWatchFolder(child) end
        end
        for _, d in ipairs(Workspace:GetDescendants()) do
            if not isPlayerDescendant(d) and (taphIsTripwire(d.Name) or taphIsTripmine(d.Name)) then
                taphCheckObject(d)
            end
        end
    end)
end

local function taphRefreshAll()
    pcall(function()
        local function scanFor(cfg, checkFn)
            for obj in pairs(cfg.tracked) do
                if not cfg.enabled then taphRemoveESP(obj); cfg.tracked[obj] = nil end
            end
            if cfg.enabled then
                for _, d in ipairs(Workspace:GetDescendants()) do
                    if not isPlayerDescendant(d) and checkFn(d.Name) and not cfg.tracked[d] then
                        cfg.tracked[d] = true
                        task.delay(0.05, function() taphApplyESP(d, cfg) end)
                    end
                end
            end
        end
        scanFor(TaphESP.tripwire, taphIsTripwire)
        scanFor(TaphESP.tripmine, taphIsTripmine)
    end)
end

local function lmsDownload(songName, callback)
    local url = LMS_TRACKS[songName]
    if not url then if callback then callback(nil) end; return nil end

    local safeName = songName:gsub("[^%w]", "_")
    local path     = LMS_FOLDER .. "/" .. safeName .. ".mp3"

    if not isfile(path) then
        local ok, data = pcall(function() return game:HttpGet(url) end)
        if not ok or not data or #data == 0 then
            if callback then callback(nil) end; return nil
        end
        pcall(function() writefile(path, data) end)
    end

    local asset
    local ok = pcall(function() asset = getcustomasset(path) end)
    if not ok or not asset then
        if callback then callback(nil) end; return nil
    end

    lmsDownloaded[songName] = asset
    if callback then callback(asset) end
    return asset
end

local function lmsFindSound()
    local function check(sound)
        if not sound or not sound:IsA("Sound") then return false end
        local n = sound.Name:lower()
        return n == "lastsurvivor" or n:find("last") or n:find("lms") or n:find("survivor")
    end

    local locs = {
        Workspace:FindFirstChild("Sounds"),
        Workspace:FindFirstChild("Themes"),
        Workspace:FindFirstChild("Music"),
        Lighting, SoundService,
    }
    for _, loc in ipairs(locs) do
        if loc then
            for _, s in ipairs(loc:GetChildren()) do
                if check(s) then return s end
            end
        end
    end

    for _, s in ipairs(Workspace:GetDescendants()) do
        if check(s) then return s end
    end

    for _, s in ipairs(ReplicatedStorage:GetDescendants()) do
        if check(s) then return s end
    end

    return nil
end

local function lmsIsActive()
    local ok, r = pcall(function()
        local pf = Workspace:FindFirstChild("Players")
        if pf then
            local sf = pf:FindFirstChild("Survivors")
            if sf then
                local alive = 0
                for _, s in ipairs(sf:GetChildren()) do
                    local h = s:FindFirstChildOfClass("Humanoid")
                    if h and h.Health > 0 then alive = alive + 1 end
                end
                if alive == 1 then return true end
                if alive > 1  then return false end
            end
        end
        local alive2 = 0
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character then
                local h = plr.Character:FindFirstChildOfClass("Humanoid")
                if h and h.Health > 0 then alive2 = alive2 + 1 end
            end
        end
        if alive2 == 1 then return true end
        local sound = lmsFindSound()
        return sound ~= nil and sound.IsPlaying
    end)
    return ok and r or false
end

local function lmsApplySong(songName)
    local sound = lmsFindSound()
    if not sound then return false end

    if not lmsOriginalId then lmsOriginalId = sound.SoundId end

    local asset = lmsDownloaded[songName] or lmsDownload(songName)
    if not asset then return false end

    pcall(function()
        sound:Stop()
        sound.SoundId = asset
        sound.Volume  = 0.55
        task.wait(0.05)
        sound:Play()
    end)
    return true
end

local function lmsResetSong()
    pcall(function()
        local sound = lmsFindSound()
        if sound and lmsOriginalId then
            sound:Stop()
            sound.SoundId = lmsOriginalId
            task.wait(0.05)
            sound:Play()
        end
    end)
end

local function lmsMonitorLoop()
    if not lmsDownloaded[lmsSelectedSong] then
        task.spawn(function() lmsDownload(lmsSelectedSong) end)
    end

    local wasActive = false

    while lmsAutoPlay do
        local active = false
        pcall(function()
            active = lmsIsActive()

            if active and not wasActive then
                wasActive = true
                lmsApplySong(lmsSelectedSong)
            elseif active and wasActive then
                local sound = lmsFindSound()
                if sound then
                    local ca = lmsDownloaded[lmsSelectedSong]
                    if ca and sound.SoundId ~= ca then
                        lmsApplySong(lmsSelectedSong)
                    end
                end
            elseif not active and wasActive then
                wasActive = false
            end
        end)

        task.wait(active and 3 or 1)
    end
end

local function lmsStart()
    if lmsThread then task.cancel(lmsThread) end
    lmsThread = task.spawn(lmsMonitorLoop)
end

local function lmsStop()
    if lmsThread then task.cancel(lmsThread); lmsThread = nil end
    lmsResetSong()
end

LocalPlayer.CharacterAdded:Connect(function()
    task.wait(3)
    if lmsAutoPlay then lmsStart() end
end)

if lmsAutoPlay then task.spawn(lmsStart) end

local KB = {
    enabled    = get("kbEnabled",     false),
    mode       = get("kbMode",        "Auto"),
    smoothness = get("kbSmooth",      8),
    prediction = get("kbPredict",     0.12),
    maxRange   = get("kbMaxRange",    120),
    fovDeg     = get("kbFOV",         360),
    autoSwing  = get("kbAutoSwing",   false),
    swingRange = get("kbSwingRange",  6),
    swingCD    = get("kbSwingCD",     1.0),
    camLock    = get("kbCamLock",     false),
    target     = nil,
    targetChar = nil,
    gyro       = nil,
    renderConn = nil,
    lastSwing  = 0,
    myHRP      = nil,
    myHum      = nil,
}

local function kbGetSurvivors()
    local out = {}
    local function tryFolder(sf)
        if not sf then return end
        for _, m in ipairs(sf:GetChildren()) do
            if m:IsA("Model") and m ~= LocalPlayer.Character then
                local hum = m:FindFirstChildOfClass("Humanoid")
                local hrp = m:FindFirstChild("HumanoidRootPart")
                if hum and hrp and hum.Health > 0 then
                    table.insert(out, {model = m, hrp = hrp, hum = hum})
                end
            end
        end
    end
    pcall(function()
        local pf = Workspace:FindFirstChild("Players")
        tryFolder(pf and pf:FindFirstChild("Survivors"))
    end)
    if #out == 0 then
        pcall(function() tryFolder(Workspace:FindFirstChild("Survivors")) end)
    end
    if #out == 0 then
        pcall(function()
            for _, plr in ipairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer and plr.Character then
                    local hum = plr.Character:FindFirstChildOfClass("Humanoid")
                    local hrp = plr.Character:FindFirstChild("HumanoidRootPart")
                    if hum and hrp and hum.Health > 0 then
                        table.insert(out, {model = plr.Character, hrp = hrp, hum = hum})
                    end
                end
            end
        end)
    end
    return out
end

local function kbInFOV(myHRP, targetHRP)
    if KB.fovDeg >= 360 then return true end
    local dir   = (targetHRP.Position - myHRP.Position).Unit
    local dot   = myHRP.CFrame.LookVector:Dot(dir)
    local angle = math.deg(math.acos(math.clamp(dot, -1, 1)))
    return angle <= KB.fovDeg / 2
end

local function kbFindNearest()
    if not KB.myHRP then return nil end
    local survivors = kbGetSurvivors()
    local best, bestD = nil, math.huge
    local myPos = KB.myHRP.Position
    for _, s in ipairs(survivors) do
        local d = (s.hrp.Position - myPos).Magnitude
        if d <= KB.maxRange and d < bestD and kbInFOV(KB.myHRP, s.hrp) then
            best, bestD = s, d
        end
    end
    return best
end

local function kbValidateTarget()
    if not KB.target or not KB.target.Parent then return false end
    local hum = KB.target.Parent:FindFirstChildOfClass("Humanoid")
    return hum ~= nil and hum.Health > 0
end

local function kbAttachGyro(hrp)
    if KB.gyro and KB.gyro.Parent then return end
    pcall(function()
        local bg = Instance.new("BodyGyro")
        bg.Name        = "_KBAimGyro"
        bg.D           = 100
        bg.P           = 1e5
        bg.MaxTorque   = Vector3.new(0, 1e6, 0)
        bg.CFrame      = hrp.CFrame
        bg.Parent      = hrp
        KB.gyro        = bg
    end)
end

local function kbDetachGyro()
    pcall(function()
        if KB.gyro and KB.gyro.Parent then KB.gyro:Destroy() end
    end)
    KB.gyro = nil
end

local function kbStop()
    if KB.renderConn then pcall(function() KB.renderConn:Disconnect() end); KB.renderConn = nil end
    kbDetachGyro()
    KB.target = nil; KB.targetChar = nil
    pcall(function()
        local c = LocalPlayer.Character
        if c then
            local h = c:FindFirstChildOfClass("Humanoid")
            if h then h.AutoRotate = true end
        end
    end)
end

local function kbStart()
    kbStop()

    KB.renderConn = RunService.RenderStepped:Connect(function()
        if not KB.enabled then return end

        local ok = pcall(function()
            local c = LocalPlayer.Character; if not c then return end
            KB.myHRP = c:FindFirstChild("HumanoidRootPart")
            KB.myHum = c:FindFirstChildOfClass("Humanoid")
        end)
        if not ok or not KB.myHRP or not KB.myHum then return end

        if KB.myHum.AutoRotate then KB.myHum.AutoRotate = false end

        kbAttachGyro(KB.myHRP)
        if not KB.gyro or not KB.gyro.Parent then return end

        if KB.mode == "Auto" then
            local s = kbFindNearest()
            if s then KB.target = s.hrp; KB.targetChar = s.model
            else KB.target = nil; KB.targetChar = nil end
        else
            if not kbValidateTarget() then
                KB.target = nil; KB.targetChar = nil
            end
        end

        if not KB.target then
            KB.gyro.CFrame = KB.myHRP.CFrame
            return
        end

        local vel = KB.target.Velocity or Vector3.zero
        local aimPos = KB.target.Position + vel * KB.prediction

        local myPos  = KB.myHRP.Position
        local toTgt  = aimPos - myPos
        local flat   = Vector3.new(toTgt.X, 0, toTgt.Z)
        if flat.Magnitude < 0.05 then return end

        local targetCF = CFrame.lookAt(myPos, myPos + flat)
        local lerpFactor = math.clamp(KB.smoothness / 50, 0.01, 1)
        KB.gyro.CFrame = KB.gyro.CFrame:Lerp(targetCF, lerpFactor)

        if KB.camLock then
            pcall(function()
                Camera.CFrame = CFrame.lookAt(
                    Camera.CFrame.Position,
                    KB.target.Position + Vector3.new(0, 1.5, 0)
                )
            end)
        end

        if KB.autoSwing then
            local now = tick()
            local dist = (KB.target.Position - myPos).Magnitude
            if dist <= KB.swingRange and now - KB.lastSwing >= KB.swingCD then
                KB.lastSwing = now
                pcall(function() fireRemote("Punch", BUF_PUNCH) end)
            end
        end
    end)
end

LocalPlayer.CharacterAdded:Connect(function()
    task.wait(1)
    KB.myHRP = nil; KB.myHum = nil; KB.gyro = nil
    KB.target = nil; KB.targetChar = nil
    if KB.enabled then kbStart() end
end)

local function espHL(color)
    local fc, ft = color, ESP.Settings.Transparency
    if ESP.Settings.Style == "Glow" then
        fc = Color3.new(math.min(color.R*1.2,1), math.min(color.G*1.2,1), math.min(color.B*1.2,1))
        ft = ft * 0.8
    elseif ESP.Settings.Style == "Outline" or ESP.Settings.Style == "Minimal" then
        ft = 1
    end
    return fc, color, ft
end

local function espAdd(obj, tag, color, isPlayer)
    pcall(function()
        if not obj or obj:FindFirstChild(tag) then return end
        local root = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Part")
            or obj:FindFirstChild("Handle") or obj.PrimaryPart
        if not root then
            for _, d in ipairs(obj:GetDescendants()) do if d:IsA("BasePart") then root = d; break end end
        end
        if not root then return end
        local fc, oc, ft = espHL(color)
        if ESP.Settings.Style ~= "Minimal" then
            local hl = Instance.new("Highlight")
            hl.Name = tag; hl.FillColor = fc; hl.FillTransparency = ft
            hl.OutlineColor = oc; hl.OutlineTransparency = 0.1
            hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
            hl.Adornee = obj; hl.Parent = obj
        end
        local bb = Instance.new("BillboardGui")
        bb.Name = tag.."_BB"; bb.Adornee = root
        bb.Size = UDim2.new(0, 140, 0, isPlayer and 32 or 28)
        bb.StudsOffset = Vector3.new(0, isPlayer and 4.5 or 3.5, 0)
        bb.AlwaysOnTop = true; bb.MaxDistance = 1000; bb.Parent = obj
        local lbl = Instance.new("TextLabel")
        lbl.Size = UDim2.new(1,0,1,0); lbl.BackgroundTransparency = 1
        lbl.TextColor3 = Color3.new(1,1,1); lbl.TextStrokeTransparency = 0.3
        lbl.TextStrokeColor3 = color; lbl.TextSize = ESP.Settings.TextSize
        lbl.Font = Enum.Font.GothamBold; lbl.Text = obj.Name; lbl.Parent = bb
        local function distText()
            if ESP.Settings.ShowDistance and LocalPlayer.Character
                and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and root then
                return string.format("  •  %.0fm",
                    (LocalPlayer.Character.HumanoidRootPart.Position - root.Position).Magnitude)
            end
            return ""
        end
        if isPlayer then
            local hum = obj:FindFirstChildOfClass("Humanoid")
            if hum and ESP.Settings.ShowHealth then
                local function upd()
                    pcall(function()
                        if not (lbl and lbl.Parent) then return end
                        lbl.Text = obj.Name..distText().."  "
                            ..math.floor((hum.Health/math.max(hum.MaxHealth,1))*100).."%"
                    end)
                end
                upd(); ESP.Conns.health[obj] = hum.HealthChanged:Connect(upd)
            else
                local function upd()
                    pcall(function() if not(lbl and lbl.Parent)then return end; lbl.Text=obj.Name..distText() end)
                end
                upd(); task.spawn(function() while obj and obj.Parent and lbl and lbl.Parent do task.wait(1);upd()end end)
            end
        else
            local prog = obj:FindFirstChild("Progress") or obj:FindFirstChild("GeneratorProgress")
            if prog and prog:IsA("NumberValue") then
                local function upd()
                    pcall(function()
                        if not(lbl and lbl.Parent)then return end
                        lbl.Text = obj.Name..distText().."  "..math.floor(prog.Value).."%"
                    end)
                end
                upd(); ESP.Conns.progress[obj] = prog.Changed:Connect(upd)
            else
                local function upd()
                    pcall(function() if not(lbl and lbl.Parent)then return end; lbl.Text=obj.Name..distText() end)
                end
                upd(); task.spawn(function() while obj and obj.Parent and lbl and lbl.Parent do task.wait(1);upd()end end)
            end
        end
    end)
end

-- -------------------------------------------------------
--  MINION ESP — c00lkidd (Pizza Delivery) + 1x1x1x1 (Rotten)
--
--  Both killers spawn minion Models inside the Killers folder
--  as children of the killer's own Model, OR as siblings.
--  c00lkidd minions: named "Minion", "Clone", "Pizza" etc., red body
--  1x1x1x1 minions: named "Minion", "Rotten", "Forsaken" etc., green
--  We scan the Killers folder children & their children, checking
--  for models that are NOT the killer themselves (no humanoid named
--  after a real player) and match minion name patterns.
-- -------------------------------------------------------
local MINION_C00L_PATTERNS = {"minion","clone","pizza","c00l","delivery"}
local MINION_1X1_PATTERNS  = {"minion","rotten","forsaken","revive","1x1","infection"}

local function isMinionC00l(name)
    name = name:lower()
    for _, p in ipairs(MINION_C00L_PATTERNS) do if name:find(p) then return true end end
    return false
end
local function isMinion1x1(name)
    name = name:lower()
    for _, p in ipairs(MINION_1X1_PATTERNS) do if name:find(p) then return true end end
    return false
end
-- Guard: don't tag real player characters (they have a player-linked humanoid)
local function isRealPlayer(model)
    local ok, r = pcall(function()
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr.Character == model then return true end
        end
        return false
    end)
    return ok and r or false
end

local function espApplyMinionC00l()
    pcall(function()
        local kf = getKillersFolder(); if not kf then return end
        local items = {}
        -- Check direct children of Killers folder AND grandchildren (minions may be nested)
        for _, child in ipairs(kf:GetChildren()) do
            if child:IsA("Model") and not isRealPlayer(child) and isMinionC00l(child.Name) then
                table.insert(items, child)

    pcall(function()
        local function rf(hl,fc,oc,ft)
            if not hl or not hl:IsA("Highlight")then return end

                local bb=obj:FindFirstChild(tag.."_BB");if bb then local l=bb:FindFirstChildOfClass("TextLabel");if l then l.TextSize=ESP.Settings.TextSize end end

        end
        local tags = {

local function safeDisconnectAll(connTable)
    -- Use type() check BEFORE indexing any properties to avoid
    -- "attempt to index nil with Disconnect" spam
    for _, c in pairs(connTable) do
        if type(c) == "userdata" then
            pcall(function() c:Disconnect() end)
        end
    end
end

local detectionCircles     = {}

local function addKillerCircle(killer)
    pcall(function()
        if not killer or not killer:FindFirstChild("HumanoidRootPart") or detectionCircles[killer] then return end
        local hrp=killer.HumanoidRootPart
        local c=Instance.new("CylinderHandleAdornment");c.Name="KillerDetectionCircle";c.Adornee=hrp
        c.Color3=ESP.Colors.Killer;c.AlwaysOnTop=true;c.ZIndex=1;c.Transparency=0.6
        c.Radius=detectionRange;c.Height=0.12
        c.CFrame=CFrame.new(0,-(hrp.Size.Y/2+0.05),0)*CFrame.Angles(math.rad(90),0,0);c.Parent=hrp
        detectionCircles[killer]=c
    end)
end
local function removeKillerCircle(killer)
    pcall(function() if detectionCircles[killer]then detectionCircles[killer]:Destroy();detectionCircles[killer]=nil end end)
end
local function refreshKillerCircles()
    pcall(function()
        local pf=Workspace:FindFirstChild("Players");if not pf then return end
        local kf=pf:FindFirstChild("Killers");if not kf then return end
        local items={};for _,k in ipairs(kf:GetChildren())do table.insert(items,k)end
        processInBatches(items,function(k)if killerCirclesVisible then addKillerCircle(k)else removeKillerCircle(k)end end,3,0.1)
    end)
end
local function updateFacingVisual(killer,visual)
    pcall(function()
        if not(killer and visual and visual.Parent)then return end
        local hrp=killer:FindFirstChild("HumanoidRootPart");if not hrp then return end
        local myRoot=LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        local inRange,facing=false,false
        if myRoot and hrp then
            inRange=(hrp.Position-myRoot.Position).Magnitude<=detectionRange
            if inRange then facing=isStrictlyFacing(myRoot,hrp,killer.Name)end
        end
        if inRange and facing then visual.Color3=Color3.fromRGB(120,255,120);visual.Transparency=0.3
        elseif inRange then visual.Color3=Color3.fromRGB(255,120,120);visual.Transparency=0.6
        else visual.Color3=Color3.fromRGB(255,255,120);visual.Transparency=0.85 end
        local frac=math.acos(math.clamp(STRICT_FACING_DOT,-1,1))/math.pi
        visual.Radius=math.max(1,detectionRange*(0.20+0.80*frac));visual.Height=0.12
        visual.CFrame=CFrame.new(0,-(hrp.Size.Y/2+0.05),-detectionRange*(0.35+0.15*frac))*CFrame.Angles(math.rad(90),0,0)
    end)
end
local function addFacingVisual(killer)
    pcall(function()
        if not killer or not killer:IsA("Model") or facingVisuals[killer] then return end
        local hrp=killer:FindFirstChild("HumanoidRootPart");if not hrp then return end
        local v=Instance.new("CylinderHandleAdornment");v.Name="FacingCheckVisual";v.Adornee=hrp
        v.AlwaysOnTop=true;v.ZIndex=2;v.Transparency=0.55;v.Color3=Color3.fromRGB(120,255,120);v.Parent=hrp
        facingVisuals[killer]=v;updateFacingVisual(killer,v)
    end)
end
local function removeFacingVisual(killer)
    pcall(function() local v=facingVisuals[killer];if v then v:Destroy();facingVisuals[killer]=nil end end)
end
local function refreshFacingVisuals()
    pcall(function()
        local pf=Workspace:FindFirstChild("Players");if not pf then return end

    end)
end

local function getAbilityRemote()
    local ok,r=pcall(function()
        for _,n in ipairs({"RemoteEvent","AbilityRemote","UseAbility","ActivateAbility"})do
            local rem=ReplicatedStorage:FindFirstChild(n);if rem and rem:IsA("RemoteEvent")then return rem end
        end
        local net=ReplicatedStorage:FindFirstChild("Modules")and ReplicatedStorage.Modules:FindFirstChild("Network")
        if net then for _,c in ipairs(net:GetChildren())do if c:IsA("RemoteEvent")then return c end end end
        return testRemote
    end);return ok and r or testRemote
end
local function fireAbility(t)
    pcall(function()
        local rem=getAbilityRemote();if not rem then return end
        if t=="Block"then rem:FireServer("UseActorAbility",{[1]=BUF_BLOCK})
        elseif t=="Punch"then rem:FireServer("UseActorAbility",{[1]=BUF_PUNCH})
        elseif t=="Charge"then rem:FireServer("UseActorAbility",{[1]=BUF_CHARGE})
        elseif t=="Clone"then rem:FireServer("UseActorAbility",{[1]=BUF_CLONE})end
    end)
    pcall(function()
        local rem=getAbilityRemote();if not rem then return end
        if t=="Block"then rem:FireServer("Block")
        elseif t=="Punch"then rem:FireServer("Punch")
        elseif t=="Charge"then rem:FireServer("Charge")
        elseif t=="Clone"then rem:FireServer("Clone")end
    end)
end
local function extractSoundId(sound)
    local ok,r=pcall(function()return tostring(sound.SoundId):match("%d+")end);return ok and r or nil
end
local function getCharFromDesc(inst)
    local ok,r=pcall(function()
        local m=inst:FindFirstAncestorOfClass("Model")
        return m and m:FindFirstChildOfClass("Humanoid")and m or nil
    end);return ok and r or nil
end
local function attemptBlock(sound)
    pcall(function()
        if not autoBlockAudioOn then return end
        if not sound or not sound:IsA("Sound")or not sound.IsPlaying then return end
        local now=tick()
        if soundBlockedUntil[sound]and now<soundBlockedUntil[sound]then return end
        if now-lastLocalBlockTime<LOCAL_BLOCK_CD then return end
        local id=extractSoundId(sound);if not id or not TRIGGER_SOUNDS[id]then return end
        local myChar=LocalPlayer.Character;if not myChar then return end
        local myRoot=myChar:FindFirstChild("HumanoidRootPart");if not myRoot then return end
        local soundPart
        if sound.Parent and sound.Parent:IsA("BasePart")then soundPart=sound.Parent
        elseif sound.Parent and sound.Parent:IsA("Attachment")and sound.Parent.Parent and sound.Parent.Parent:IsA("BasePart")then soundPart=sound.Parent.Parent end
        if not soundPart then return end
        local char=getCharFromDesc(soundPart);if not char then return end
        local plr=Players:GetPlayerFromCharacter(char);if not plr or plr==LocalPlayer then return end
        local hrp=char:FindFirstChild("HumanoidRootPart");if not hrp then return end
        if(hrp.Position-myRoot.Position).Magnitude>detectionRange then return end
        if not isStrictlyFacing(myRoot,hrp,char.Name)then return end
        if blockdelay>0 then task.wait(blockdelay)end
        if autoblocktype=="Block"then fireAbility("Block");if doubleblocktech then task.wait(0.05);fireAbility("Punch")end
        elseif autoblocktype=="Charge"then fireAbility("Charge")
        elseif autoblocktype=="7n7 Clone"then fireAbility("Clone")end
        lastLocalBlockTime=now;soundBlockedUntil[sound]=now+1.0
    end)
end
local function hookSound(sound)
    pcall(function()
        if not sound or not sound:IsA("Sound")or soundHooks[sound]then return end
        local pc=sound.Played:Connect(function()safeSpawn(attemptBlock,sound)end)
        local ic=sound:GetPropertyChangedSignal("IsPlaying"):Connect(function()if sound.IsPlaying then safeSpawn(attemptBlock,sound)end end)
        local dc=sound.Destroying:Connect(function()
            pcall(function() pc:Disconnect() end)
            pcall(function() ic:Disconnect() end)
            pcall(function() dc:Disconnect() end)
            soundHooks[sound]=nil;soundBlockedUntil[sound]=nil
        end)
        soundHooks[sound]={pc,ic,dc};if sound.IsPlaying then safeSpawn(attemptBlock,sound)end
    end)
end
local function setupSoundHooks()
    pcall(function()
        local pf=Workspace:FindFirstChild("Players");if not pf then return end
        local kf=pf:FindFirstChild("Killers");if not kf then return end
        local items={};for _,d in pairs(kf:GetDescendants())do if d:IsA("Sound")then table.insert(items,d)end end
        processInBatches(items,function(d)hookSound(d)end,5,0.1)
        kf.DescendantAdded:Connect(function(d)if d:IsA("Sound")then task.delay(0.2,function()hookSound(d)end)end end)
    end)
end

local genFlowSolverEnabled = get("genFlowSolverEnabled", false)
local genFlowNodeDelay     = get("genFlowNodeDelay",     0.04)
local genFlowLineDelay     = get("genFlowLineDelay",     0.60)
local function genGetDir(cr,cc,or2,oc)
    if or2<cr then return"up"elseif or2>cr then return"down"elseif oc<cc then return"left"else return"right"end
end
local function genGetConns(prev,curr,nxt)
    local conn={}
    if prev and curr then
        local d=genGetDir(curr.row,curr.col,prev.row,prev.col)
        if d=="up"then d="down"elseif d=="down"then d="up"elseif d=="left"then d="right"else d="left"end;conn[d]=true
    end
    if nxt and curr then local d=genGetDir(curr.row,curr.col,nxt.row,nxt.col);if d then conn[d]=true end end;return conn
end
local function genIsNB(r1,c1,r2,c2)
    if r2==r1-1 and c2==c1 then return"up"end;if r2==r1+1 and c2==c1 then return"down"end
    if r2==r1 and c2==c1-1 then return"left"end;if r2==r1 and c2==c1+1 then return"right"end;return false
end
local function genKey(n)return n.row.."-"..n.col end
local function genOrderPath(path,endpoints)
    if not path or #path==0 then return path end
    local inPath={};for _,n in ipairs(path)do inPath[genKey(n)]=n end
    local startNode
    for _,ep in ipairs(endpoints or{})do
        for _,n in ipairs(path)do if n.row==ep.row and n.col==ep.col then startNode={row=ep.row,col=ep.col};break end end
        if startNode then break end
    end
    if not startNode then
        for _,n in ipairs(path)do
            local nb=0;for _,d in ipairs({{-1,0},{1,0},{0,-1},{0,1}})do if inPath[(n.row+d[1]).."-"..(n.col+d[2])]then nb=nb+1 end end
            if nb==1 then startNode={row=n.row,col=n.col};break end
        end
    end
    if not startNode then startNode={row=path[1].row,col=path[1].col}end
    local remaining,ordered={},{}
    for _,n in ipairs(path)do remaining[genKey(n)]={row=n.row,col=n.col}end
    local current=startNode;table.insert(ordered,{row=current.row,col=current.col});remaining[genKey(current)]=nil
    local iter=0
    while next(remaining)and iter<100 do
        iter=iter+1;local found=false
        for key,node in pairs(remaining)do
            if genIsNB(current.row,current.col,node.row,node.col)then
                table.insert(ordered,{row=node.row,col=node.col});remaining[key]=nil;current=node;found=true;break
            end
        end;if not found then break end
    end;return ordered
end
local function genDrawSolution(puzzle)
    pcall(function()
        if not puzzle or not puzzle.Solution then return end
        local indices={};for i=1,#puzzle.Solution do table.insert(indices,i)end
        for i=#indices,2,-1 do local j=math.random(1,i);indices[i],indices[j]=indices[j],indices[i]end
        for _,ci in ipairs(indices)do
            local path=puzzle.Solution[ci];local ordered=genOrderPath(path,puzzle.targetPairs[ci]);puzzle.paths[ci]={}
            for i,node in ipairs(ordered)do
                table.insert(puzzle.paths[ci],{row=node.row,col=node.col})
                local conn=genGetConns(ordered[i-1],node,ordered[i+1])
                puzzle.gridConnections=puzzle.gridConnections or{};puzzle.gridConnections[genKey(node)]=conn
                pcall(function()puzzle:updateGui()end);task.wait(genFlowNodeDelay)
            end;task.wait(genFlowLineDelay);pcall(function()puzzle:checkForWin()end)
        end
    end)
end

    end
end

local function startAimPunch()
    pcall(function()
        if not aimPunchEnabled then return end
        local myChar=LocalPlayer.Character;if not myChar then return end
        local myRoot=myChar:FindFirstChild("HumanoidRootPart");if not myRoot then return end
        local kf=Workspace:FindFirstChild("Players")and Workspace.Players:FindFirstChild("Killers");if not kf then return end
        local nearest,nDist=nil,math.huge
        for _,killer in pairs(kf:GetChildren())do
            local hrp=killer:FindFirstChild("HumanoidRootPart")
            if hrp then local d=(hrp.Position-myRoot.Position).Magnitude;if d<=lockMaxDistance and d<nDist then nDist=d;nearest=killer end end
        end
        if not nearest then return end
        local tHRP=nearest:FindFirstChild("HumanoidRootPart");if not tHRP then return end
        local hum=myChar:FindFirstChildOfClass("Humanoid");if hum then hum.AutoRotate=false end
        task.spawn(function()
            local st=tick()
            while tick()-st<aimWindow do
                pcall(function()
                    local c=LocalPlayer.Character;local r=c and c:FindFirstChild("HumanoidRootPart")
                    if not r or not tHRP or not tHRP.Parent then return end
                    local vel=tHRP.Velocity or Vector3.zero;local vm=vel.Magnitude
                    local pd=vm>0.5 and vel.Unit or tHRP.CFrame.LookVector
                    if pd~=pd then pd=Vector3.zero end
                    r.CFrame=CFrame.lookAt(r.Position,tHRP.Position+(pd*predictionValue)+(vel*0.1))
                end);task.wait()
            end
            local c=LocalPlayer.Character;if c then local h=c:FindFirstChildOfClass("Humanoid");if h then h.AutoRotate=true end end
        end)
    end)
end

-- ============================================================
--  HEARTBEAT LOOP
-- ============================================================
local _lastLockScan=0;local _lastAimScan=0;local _lastPunchScan=0;local hbIter=0
RunService.Heartbeat:Connect(function()
    pcall(function()
        hbIter=hbIter+1;if hbIter>10000 then hbIter=0 end;local now=tick()
        if characterLockOn and(now-_lastLockScan)>=0.1 then
            _lastLockScan=now;if not cachedAnimator then refreshAnimator()end
            local anim=cachedAnimator;if anim and anim.Parent then
                for _,track in ipairs(anim:GetPlayingAnimationTracks())do
                    local id=tostring(track.Animation and track.Animation.AnimationId or""):match("%d+")
                    if id and punchAnimSet[id]then safeSpawn(startAimPunch);break end
                end
            end
        end
        if aimPunchEnabled and(now-_lastAimScan)>=0.1 then
            _lastAimScan=now;if not cachedAnimator then refreshAnimator()end
            local anim=cachedAnimator;if anim then
                for _,track in ipairs(anim:GetPlayingAnimationTracks())do
                    local id=tostring(track.Animation and track.Animation.AnimationId or""):match("%d+")
                    if id and punchAnimSet[id]then
                        local last=lastAimTrigger[track]
                        if not last or(now-last)>=aimCooldown then
                            if(track.TimePosition or 0)<=0.1 then lastAimTrigger[track]=now;safeSpawn(startAimPunch)end
                        end
                    end
                end
            end
        end
        if autoPunchOn and(now-_lastPunchScan)>=0.2 then
            _lastPunchScan=now;local myChar=LocalPlayer.Character;if myChar then
                local myRoot=myChar:FindFirstChild("HumanoidRootPart");if myRoot then
                    local pf=Workspace:FindFirstChild("Players");if pf then
                        local kf=pf:FindFirstChild("Killers");if kf and hbIter%2==0 then
                            for _,name in ipairs(killerNames)do
                                local killer=kf:FindFirstChild(name);if killer then
                                    local r=killer:FindFirstChild("HumanoidRootPart")
                                    if r and(r.Position-myRoot.Position).Magnitude<=10 then
                                        fireRemote("Punch",BUF_PUNCH);break
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
        -- AUTO CROUCH (every 0.1s via heartbeat)
        -- Fires UseActorAbility Crouch as primary (matches how Block/Punch/Charge work).
        -- Falls back to plain FireServer("Crouch") if Crouching attribute not set.
        -- Also satisfies V1PRSTAB Dash-Stab Tech which checks GetAttribute("Crouching").
        if autoCrouchEnabled and (now - _lastCrouchFire) >= 0.1 then
            _lastCrouchFire = now
            pcall(function()
                if testRemote then
                    testRemote:FireServer("UseActorAbility", {[1] = BUF_CROUCH})
                end
            end)
            -- Backup: if attribute still not set after primary, send plain remote too
            pcall(function()
                local char = LocalPlayer.Character
                if char and char:GetAttribute("Crouching") ~= true and testRemote then
                    testRemote:FireServer("Crouch")
                end
            end)
        end
    end)
end)
