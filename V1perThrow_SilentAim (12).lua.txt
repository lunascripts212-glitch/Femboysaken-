-- ────────────────────────────────────────────────
-- Killer tracking
-- ────────────────────────────────────────────────
local killerMotionData = {}

local function getKillerVelocity(hrp)
    local now = tick()
    local pos = hrp.Position
    local data = killerMotionData[hrp] or {lastPos = pos, lastTime = now, velocity = Vector3.zero}
    killerMotionData[hrp] = data

    local dt = now - data.lastTime
    if dt > 0 then
        data.velocity = (pos - data.lastPos) / dt
        data.lastPos = pos
        data.lastTime = now
    end
    return data.velocity
end

local function getNearestKiller(fromPos)
    local killers = workspace:FindFirstChild("Players") 
        and workspace.Players:FindFirstChild("Killers")
    if not killers then return nil end

    local nearest, best = nil, math.huge
    for _, model in killers:GetChildren() do
        local hrp = model:FindFirstChild("HumanoidRootPart")
        local hum = model:FindFirstChildOfClass("Humanoid")
        if hrp and hum and hum.Health > 0 then
            local dist = (hrp.Position - fromPos).Magnitude
            if dist < best then
                best = dist
                nearest = model
            end
        end
    end
    return nearest
end

-- ────────────────────────────────────────────────
-- Ballistic camera spoof (head aim)
-- ────────────────────────────────────────────────
local function buildSpoofCF(myHRP, killerHRP, abilityCfg)
    local vel = getKillerVelocity(killerHRP)
    local v0 = abilityCfg.MaxSpeed
    local g = abilityCfg.ProjectileArc

    local hum = myHRP.Parent:FindFirstChildOfClass("Humanoid")
    local hipH = hum and hum.HipHeight or 1.35
    local spawnY = (hipH + myHRP.Size.Y / 2) / 2
    local spawnPos = myHRP.Position + Vector3.new(0, spawnY, 0)

    local predicted = killerHRP.Position + vel * PREDICTION
    local target = predicted + Vector3.new(0, 1.4, 0)  -- head approx

    local delta = target - spawnPos
    local flat = Vector3.new(delta.X, 0, delta.Z)
    local dx = flat.Magnitude
    local dy = delta.Y

    if dx < 0.1 then
        local dir = dy >= 0 and Vector3.yAxis or -Vector3.yAxis
        return CFrame.new(Camera.CFrame.Position, Camera.CFrame.Position + dir)
    end

    local flatUnit = flat.Unit
    local vv = v0 * v0
    local disc = vv * vv - g * (g * dx * dx + 2 * dy * vv)
    local tanTheta = (disc < 0) and (dy / dx) or ((vv - math.sqrt(disc)) / (g * dx))

    local alpha = math.atan2(3 * tanTheta - 1, 3 + tanTheta)

    local yaw = CFrame.lookAt(Camera.CFrame.Position, Camera.CFrame.Position + flatUnit)
    return yaw * CFrame.Angles(alpha, 0, 0)
end

-- ────────────────────────────────────────────────
-- Patch crystal ability (only spoof on actual fire)
-- ────────────────────────────────────────────────
local patched = false
local originalCB

local function getLocalActor()
    for _, a in Actors.CurrentActors do
        if a.Player == LocalPlayer then return a end
    end
    return nil
end

local function applyPatch(actor)
    if patched or not actor or not actor.Behavior or not actor.Behavior.Abilities or not actor.Behavior.Abilities.Crystal then
        return
    end

    local NetworkRF = ReplicatedStorage.Modules.Network:FindFirstChild("RemoteFunction")

    originalCB = actor.Behavior.Abilities.Crystal.Callback

    actor.Behavior.Abilities.Crystal.Callback = function(self, input)
        if RunService:IsServer() then
            return originalCB(self, input)
        end

        -- Let normal equip / channel / charge happen
        local res = originalCB(self, input)

        -- Only spoof when user actually fires (input == 1)
        if input == 1 and _G.LoveThrow_Enabled then
            local myHRP = self.Rig and self.Rig:FindFirstChild("HumanoidRootPart")
            if not myHRP then return res end

            local killer = getNearestKiller(myHRP.Position)
            local killerHRP = killer and killer:FindFirstChild("HumanoidRootPart")
            if not killerHRP then return res end

            if isMobile and NetworkRF then
                local origInvoke = getcallbackvalue and getcallbackvalue(NetworkRF, "OnClientInvoke")
                local origDev = LocalPlayer:GetAttribute("Device") or "PC"

                LocalPlayer:SetAttribute("Device", "Mobile")

                local hook = function(name, ...)
                    if name == "GetCameraCF" then
                        NetworkRF.OnClientInvoke = origInvoke
                        return buildSpoofCF(myHRP, killerHRP, self.Config.Crystal)
                    end
                    return origInvoke and origInvoke(name, ...) or nil
                end
                NetworkRF.OnClientInvoke = hook

                -- Safety restore
                task.delay(2, function()
                    if NetworkRF and getcallbackvalue and getcallbackvalue(NetworkRF, "OnClientInvoke") == hook then
                        NetworkRF.OnClientInvoke = origInvoke
                    end
                    LocalPlayer:SetAttribute("Device", origDev)
                end)
            end
            -- PC: no extra spoof needed if server uses client camera (common in these games)
        end

        return res
    end

    patched = true
    print("[LoveThrow] Patched — manual throw with silent head aim")
    WindUI:Notify({ Title = "LoveThrow", Content = "Patched — throw manually", Duration = 3.5 })
end

local function removePatch()
    if not patched then return end
    local actor = getLocalActor()
    if actor and actor.Behavior and actor.Behavior.Abilities and actor.Behavior.Abilities.Crystal and originalCB then
        actor.Behavior.Abilities.Crystal.Callback = originalCB
    end
    patched = false
end

-- ────────────────────────────────────────────────
-- Actor / round watcher
-- ────────────────────────────────────────────────
_G.LoveThrow_Enabled = true
_G.LoveThrow_Unloaded = false

task.spawn(function()
    local lastActor = nil
    while not _G.LoveThrow_Unloaded do
        task.wait(0.6)
        local actor = getLocalActor()

        if actor ~= lastActor then
            killerMotionData = {}
            if lastActor then
                removePatch()
                WindUI:Notify({ Title = "LoveThrow", Content = "New round — repatching", Duration = 3 })
            end
            lastActor = actor
            if actor then
                applyPatch(actor)
            end
        end
    end
    removePatch()
end)

print("[LoveThrow] Loaded — WindUI version — charge & throw yourself")
