-- ────────────────────────────────────────────────
-- KILLER FIND + VELOCITY
-- ────────────────────────────────────────────────
local killerMotionData = {}

local function getKillerVelocity(hrp)
    local now = tick()
    local pos = hrp.Position
    local data = killerMotionData[hrp]
    if not data then
        killerMotionData[hrp] = {lastPos = pos, lastTime = now, velocity = Vector3.zero}
        return Vector3.zero
    end
    local dt = now - data.lastTime
    if dt <= 0 then return data.velocity end
    local vel = (pos - data.lastPos) / dt
    data.lastPos = pos
    data.lastTime = now
    data.velocity = vel
    return vel
end

local function getNearestKiller(fromPos)
    local killersFolder = workspace:FindFirstChild("Players") 
        and workspace.Players:FindFirstChild("Killers")
    if not killersFolder then return nil end

    local nearest, bestDist = nil, math.huge
    for _, model in killersFolder:GetChildren() do
        local hrp = model:FindFirstChild("HumanoidRootPart")
        local hum = model:FindFirstChildOfClass("Humanoid")
        if hrp and hum and hum.Health > 0 then
            local dist = (hrp.Position - fromPos).Magnitude
            if dist < bestDist then
                bestDist = dist
                nearest = model
            end
        end
    end
    return nearest
end

-- ────────────────────────────────────────────────
-- SPOOF CAMERA CFRAME (mobile / server checks UpVector/3 too)
-- ────────────────────────────────────────────────
local function buildHeadAimSpoofCF(myHRP, killerHRP, abilityCfg)
    local vel = getKillerVelocity(killerHRP)
    local v0 = abilityCfg.MaxSpeed
    local g  = abilityCfg.ProjectileArc

    local hipHeight = (myHRP.Parent:FindFirstChildOfClass("Humanoid") and .HipHeight) or 1.35
    local spawnOffset = (hipHeight + myHRP.Size.Y / 2) / 2
    local spawnPos = myHRP.Position + Vector3.new(0, spawnOffset, 0)

    local predictedPos = killerHRP.Position + vel * PREDICTION
    local targetPos = predictedPos + Vector3.new(0, 1.4, 0)   -- ≈ head height

    local delta = targetPos - spawnPos
    local flat = Vector3.new(delta.X, 0, delta.Z)
    local dx = flat.Magnitude
    local dy = delta.Y

    if dx < 0.1 then
        local dir = dy >= 0 and Vector3.yAxis or -Vector3.yAxis
        return CFrame.new(Camera.CFrame.Position, Camera.CFrame.Position + dir)
    end

    local flatUnit = flat.Unit

    local vv = v0 * v0
    local disc = vv*vv - g*(g*dx*dx + 2*dy*vv)
    local tanTheta
    if disc < 0 then
        tanTheta = dy / dx
    else
        tanTheta = (vv - math.sqrt(disc)) / (g * dx)
    end

    local alpha = math.atan2(3*tanTheta - 1, 3 + tanTheta)

    local yawCF = CFrame.lookAt(Camera.CFrame.Position, Camera.CFrame.Position + flatUnit)
    return yawCF * CFrame.Angles(alpha, 0, 0)
end

-- ────────────────────────────────────────────────
-- PATCH CRYSTAL ABILITY
-- ────────────────────────────────────────────────
local patched = false
local originalCrystalCallback

local function getLocalActor()
    for _, actor in Actors.CurrentActors do
        if actor.Player == LocalPlayer then
            return actor
        end
    end
    return nil
end

local function applyPatch(actor)
    if patched then return end
    if not actor or not actor.Behavior or not actor.Behavior.Abilities or not actor.Behavior.Abilities.Crystal then
        return
    end

    local NetworkRF = ReplicatedStorage.Modules.Network:FindFirstChild("RemoteFunction")

    originalCrystalCallback = actor.Behavior.Abilities.Crystal.Callback

    actor.Behavior.Abilities.Crystal.Callback = function(self, arg)
        if RunService:IsServer() then
            return originalCrystalCallback(self, arg)
        end

        -- Let normal channel/charge happen
        local result = originalCrystalCallback(self, arg)

        -- We only want to spoof when actually firing (input == 1)
        if arg == 1 and _G.LoveThrow_Enabled then
            local myHRP = self.Rig and self.Rig:FindFirstChild("HumanoidRootPart")
            if not myHRP then return result end

            local killerModel = getNearestKiller(myHRP.Position)
            local killerHRP = killerModel and killerModel:FindFirstChild("HumanoidRootPart")
            if not killerHRP then return result end

            -- Mobile spoof path
            if isMobile and NetworkRF then
                local origInvoke = getcallbackvalue(NetworkRF, "OnClientInvoke")
                local origDevice = LocalPlayer:GetAttribute("Device")

                LocalPlayer:SetAttribute("Device", "Mobile")

                NetworkRF.OnClientInvoke = function(reqName, ...)
                    if reqName == "GetCameraCF" then
                        NetworkRF.OnClientInvoke = origInvoke
                        return buildHeadAimSpoofCF(myHRP, killerHRP, self.Config.Crystal)
                    end
                    if origInvoke then
                        return origInvoke(reqName, ...)
                    end
                end

                -- Give attribute time to replicate
                task.wait(0.08)

                -- Let normal fire go through
                -- (we already returned original result earlier — this is just spoof cleanup)
                task.delay(1.5, function()
                    if NetworkRF and getcallbackvalue(NetworkRF, "OnClientInvoke") ~= origInvoke then
                        NetworkRF.OnClientInvoke = origInvoke
                    end
                    LocalPlayer:SetAttribute("Device", origDevice or "PC")
                end)
            end
            -- PC: no mouse movement — we rely purely on GetCameraCF spoof (most games accept it)
        end

        return result
    end

    patched = true
    Rayfield:Notify({
        Title = "LoveThrow",
        Content = "Crystal patched — silent aim active on fire",
        Duration = 4.2,
    })
end

local function removePatch()
    if not patched then return end
    local actor = getLocalActor()
    if actor and actor.Behavior and actor.Behavior.Abilities and actor.Behavior.Abilities.Crystal and originalCrystalCallback then
        actor.Behavior.Abilities.Crystal.Callback = originalCrystalCallback
    end
    patched = false
end

-- ────────────────────────────────────────────────
-- ACTOR / ROUND WATCHER
-- ────────────────────────────────────────────────
_G.LoveThrow_Enabled = true
_G.LoveThrow_Unloaded = false

task.spawn(function()
    local lastActor = nil

    while not _G.LoveThrow_Unloaded do
        task.wait(0.8)

        local actor = getLocalActor()

        if actor ~= lastActor then
            killerMotionData = {}
            if lastActor then
                removePatch()
                Rayfield:Notify({
                    Title = "LoveThrow",
                    Content = "New round — re-patching crystal ability",
                    Duration = 3.5,
                })
            end

            lastActor = actor

            if actor then
                applyPatch(actor)
            end
        end
    end

    removePatch()
end)

print("[LoveThrow] Loaded — charge & throw manually, aim silently goes to killer head")
