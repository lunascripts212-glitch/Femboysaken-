local LERP_SPEED          = 0.37 * 1.6
local AIM_SNAP_DELAY      = 0.25
local CHECK_INTERVAL      = 0.05

-- Ping calibration
local measuredPingMs  = 80
local pingLabel       = nil
local autoPingEnabled = true

local function measurePing()
    local ok, ping = return statsService.Network.ServerStatsItem["Data Ping"]:GetValue()
    if ok and type(ping) == "number" and ping > 0 then return ping end
    local ok2, ping2 = return clientPlayer:GetNetworkPing() * 1000
    if ok2 and type(ping2) == "number" and ping2 > 0 then return ping2 end
    return nil
end

task.spawn(function()
    while isRunning do
        task.wait(2)
        if not autoPingEnabled then continue end
        local ping = measurePing()
        if ping then
            measuredPingMs = ping
            local pingSec = math.clamp(ping / 1000, 0.01, 0.40)
            DASH_FLIP_DELAY = SERVER_WINDUP + pingSec
            if pingLabel then
                pingLabel:SetDescription(string.format("Ping: %dms  |  Flip Delay: %.3fs", math.floor(ping), DASH_FLIP_DELAY))
            end
        end
    end
end)

local function syncLungeHoldToPosture()
    LUNGE_HOLD_DURATION = autoCrouchEnabled and HITBOX_TIME_CROUCHED or HITBOX_TIME_UNCROUCHED
end

local CROUCH_REMOTE_PRIMARY = "TwoTimeCrouching"
local CROUCH_REMOTE_BACKUP  = "Crouch"
local CROUCH_FIRE_INTERVAL  = 0.1

local function getForsakenRemote()
    local rs = game:GetService("ReplicatedStorage")
    local modules = rs:FindFirstChild("Modules"); if not modules then return nil end
    local network = modules:FindFirstChild("Network"); if not network then return nil end
    return network:FindFirstChild("RemoteEvent")
end

local function isCrouching()
    local char = clientPlayer.Character; if not char then return false end
    return char:GetAttribute("Crouching") == true
end

local function fireCrouchRemote(remoteName)
    local remote = getForsakenRemote()
    if remote then remote:FireServer(remoteName) end
end

task.spawn(function()
    while isRunning do
        task.wait(CROUCH_FIRE_INTERVAL)
        if not autoCrouchEnabled or not isRunning then continue end
        fireCrouchRemote(CROUCH_REMOTE_PRIMARY)
        task.wait(0.05)
        if not isCrouching() then fireCrouchRemote(CROUCH_REMOTE_BACKUP) end
    end
end)

local RING_COLOR_BACKSTAB_IDLE = Color3.fromRGB(0,   180, 255)
local RING_COLOR_DASH_IDLE     = Color3.fromRGB(255, 140, 0)

local function getCharacter()
    return clientPlayer.Character or clientPlayer.CharacterAdded:Wait()
end

local function getDaggerButton()
    local pg = clientPlayer:FindFirstChild("PlayerGui"); if not pg then return nil end
    local mainUI = pg:FindFirstChild("MainUI"); if not mainUI then return nil end
    local container = mainUI:FindFirstChild("AbilityContainer"); if not container then return nil end
    return container:FindFirstChild("Dagger")
end

local function getDaggerCooldown()
    local btn = getDaggerButton(); if not btn then return nil end
    return btn:FindFirstChild("CooldownTime") or btn:FindFirstChild("Cooldown")
        or btn:FindFirstChildWhichIsA("NumberValue") or btn:FindFirstChildWhichIsA("StringValue")
        or btn:FindFirstChild("CooldownLabel") or btn:FindFirstChild("Timer") or btn:FindFirstChild("CD")
end

local function readCooldownValue(cdObj)
    if not cdObj then return nil end
    if cdObj:IsA("NumberValue") then return cdObj.Value end
    if cdObj:IsA("StringValue") then return tonumber(cdObj.Value) end
    if cdObj:IsA("TextLabel") or cdObj:IsA("TextBox") then return tonumber(cdObj.Text) end
    if type(cdObj.Value) == "number" then return cdObj.Value end
    if type(cdObj.Value) == "string" then return tonumber(cdObj.Value) end
    if cdObj.Text ~= nil then return tonumber(cdObj.Text) end
    return nil
end

local function getKillerSpeed(killer)
    local khrp = killer:FindFirstChild("HumanoidRootPart"); if not khrp then return 0 end
    local now = os.clock(); local cache = killerSpeedCache[killer]; local speed = 0
    if cache then
        local dt = now - cache.t
        if dt > 0 then
            local dp = khrp.Position - cache.pos
            speed = Vector3.new(dp.X, 0, dp.Z).Magnitude / dt
        end
    end
    killerSpeedCache[killer] = { pos = khrp.Position, t = now }
    return speed
end

local function getKillersFolder()
    local pf = workspaceService:FindFirstChild("Players"); if not pf then return nil end
    return pf:FindFirstChild("Killers")
end

local function isValidKillerModel(model)
    if not model then return false end
    local hrp = model:FindFirstChild("HumanoidRootPart")
    local hum = model:FindFirstChildWhichIsA("Humanoid")
    return hrp and hum and hum.Health and hum.Health > 0
end

local function tryActivateButton(btn)
    if not btn then return false end
    if btn.Activate then btn:Activate() end
    local ok, conns = pcall(function()
        if type(getconnections) == "function" and btn.MouseButton1Click then
            return getconnections(btn.MouseButton1Click)
        end
        return nil
    end)
    if ok and conns then
        for _, conn in ipairs(conns) do
            if conn.Function then conn.Function()
            elseif conn.func then conn.func()
            elseif conn.Fire then conn.Fire() end
        end
    end
    if btn.Activated then btn.Activated:Fire() end
    return true
end

local function setAutoRotate(value)
    local char = clientPlayer.Character; if not char then return end
    local hum = char:FindFirstChildWhichIsA("Humanoid")
    if hum then hum.AutoRotate = value end
end

local function bumpTriggerCount()
    totalTriggers += 1
    if triggerLabel then
        triggerLabel:SetDescription("Fires this session: " .. totalTriggers)
    end
end

local function aimAtKiller(killerModel, duration)
    if not killerModel or not isRunning then return end
    local char = getCharacter()
    local hum  = char and char:FindFirstChildWhichIsA("Humanoid")
    local hrp  = char and char:FindFirstChild("HumanoidRootPart")
    local khrp = killerModel:FindFirstChild("HumanoidRootPart")
    if not hum or not hrp or not khrp then return end
    aimRefCount += 1
    if aimRefCount == 1 then hum.AutoRotate = false end
    local function finish()
        aimRefCount = math.max(0, aimRefCount - 1)
        if aimRefCount == 0 then setAutoRotate(true) end
    end
    local t0 = os.clock(); local conn
    conn = runService.Heartbeat:Connect(function()
        if not isRunning or os.clock() - t0 >= duration then conn:Disconnect(); finish(); return end
        if khrp and hrp then
            local tgt = Vector3.new(khrp.Position.X, hrp.Position.Y, khrp.Position.Z)
            hrp.CFrame = hrp.CFrame:Lerp(CFrame.new(hrp.Position, tgt), LERP_SPEED)
        end
    end)
end

local function performDashStabTech(killerModel)
    if not killerModel or not isRunning then return end
    local char = getCharacter()
    local hrp  = char and char:FindFirstChild("HumanoidRootPart")
    local khrp = killerModel:FindFirstChild("HumanoidRootPart")
    if not hrp or not khrp then return end
    setAutoRotate(false)
    hrp.CFrame = CFrame.lookAt(hrp.Position, Vector3.new(khrp.Position.X, hrp.Position.Y, khrp.Position.Z), Vector3.new(0,1,0))
    if daggerEnabled then tryActivateButton(getDaggerButton()) end
    task.wait(DASH_FLIP_DELAY)
    if not isRunning then setAutoRotate(true); return end
    local holdDuration = LUNGE_HOLD_DURATION; local holdStart = os.clock(); local firstFrame = true; local holdConn
    holdConn = runService.Heartbeat:Connect(function()
        if not isRunning or os.clock() - holdStart >= holdDuration then
            holdConn:Disconnect(); setAutoRotate(true); return
        end
        local khrp2 = killerModel:FindFirstChild("HumanoidRootPart")
        local char2 = getCharacter(); local hrp2 = char2 and char2:FindFirstChild("HumanoidRootPart")
        if khrp2 and hrp2 then
            local look = khrp2.CFrame.LookVector; local nudgedPos = hrp2.Position
            if firstFrame then
                firstFrame = false
                nudgedPos = Vector3.new(hrp2.Position.X + look.X * DEPTH_OFFSET, hrp2.Position.Y, hrp2.Position.Z + look.Z * DEPTH_OFFSET)
            end
            hrp2.CFrame = CFrame.lookAt(nudgedPos, nudgedPos + look, Vector3.new(0,1,0))
        end
    end)
end

local radiusParts = {}
local showRadius  = true

local function getOrCreateRadiusPart(killer)
    if radiusParts[killer] then return radiusParts[killer] end
    local part = Instance.new("Part")
    part.Name = "FemboyStabRadius"; part.Anchored = true; part.CanCollide = false
    part.CanQuery = false; part.CanTouch = false; part.CastShadow = false
    part.Shape = Enum.PartType.Cylinder; part.Material = Enum.Material.Neon
    part.Transparency = 0.65; part.Color = RING_COLOR_BACKSTAB_IDLE
    part.Size = Vector3.new(0.15, 2, 2); part.Parent = workspaceService
    radiusParts[killer] = part
    killer.AncestryChanged:Connect(function()
        if not killer:IsDescendantOf(workspaceService) then
            part:Destroy(); radiusParts[killer] = nil
        end
    end)
    return part
end

local function destroyAllRadiusParts()
    for _, part in pairs(radiusParts) do part:Destroy() end
    radiusParts = {}
end

local function updateRadiusPart(killer, khrp, range, twoTimeInside)
    if not showRadius then
        local ex = radiusParts[killer]; if ex then ex:Destroy(); radiusParts[killer] = nil end; return
    end
    local part = getOrCreateRadiusPart(killer)
    local diameter = range * 2; local footY = khrp.Position.Y - 3
    part.Size = Vector3.new(0.15, diameter, diameter)
    part.CFrame = CFrame.new(khrp.Position.X, footY, khrp.Position.Z) * CFrame.Angles(0, 0, math.pi / 2)
    local isDash = (mode == "Dash-Stab Tech")
    if twoTimeInside then
        local pulse = 0.5 + 0.5 * math.sin(os.clock() * 8)
        part.Color = Color3.fromRGB(255, math.floor(pulse * (isDash and 60 or 80)), 0)
        part.Transparency = 0.3 + pulse * 0.2
    else
        part.Color = isDash and RING_COLOR_DASH_IDLE or RING_COLOR_BACKSTAB_IDLE
        part.Transparency = 0.65
    end
end

-- ============================================================
--  SINGLE HEARTBEAT LOOP
-- ============================================================
local hbIter = 0
local _lastPingSample = 0

RunService.Heartbeat:Connect(function()
    hbIter = hbIter + 1
    local now = tick()

    -- PING CALIBRATION (every 2s)
    if autoPingEnabled and (now - _lastPingSample) >= 2 then
        _lastPingSample = now
        task.spawn(function()
            local ping = measurePing()
            if ping then
                measuredPingMs = ping
                local pingSec = math.clamp(ping / 1000, 0.01, 0.40)
                DASH_FLIP_DELAY = SERVER_WINDUP + pingSec
                if pingLabel then
                    pingLabel:SetDescription(string.format("Ping: %dms  |  Flip Delay: %.3fs", math.floor(ping), DASH_FLIP_DELAY))
                end
            end
        end

        -- AUTO CROUCH (every 6 frames ≈ 0.1s)
        -- Fires UseActorAbility Crouch matching how lovesaken fires Block/Punch/Charge.
        -- Backup plain FireServer if Crouching attribute not set after primary.
        -- This also satisfies the Dash-Stab Tech crouching gate below.
        if autoCrouchEnabled and hbIter % 6 == 0 then
            local remote = getForsakenRemote()
            if remote then remote:FireServer("UseActorAbility", {[1] = "Crouch"}) end
        end
        if autoCrouchEnabled and hbIter % 12 == 0 then
            local char = clientPlayer.Character
            if char and char:GetAttribute("Crouching") ~= true then
                local remote = getForsakenRemote()
                if remote then remote:FireServer("Crouch") end
            end
        end

        -- MAIN DETECTION (only when enabled)
        if not enabled then return end

        local killersFolder = getKillersFolder(); if not killersFolder then return end
        local char = clientPlayer.Character; if not char then return end
        local hrp  = char:FindFirstChild("HumanoidRootPart"); if not hrp then return end

        for _, killer in pairs(killersFolder:GetChildren()) do
            if not isValidKillerModel(killer) then continue end
            local khrp     = killer:FindFirstChild("HumanoidRootPart")
            local dist     = (khrp.Position - hrp.Position).Magnitude
            local kSpeed   = getKillerSpeed(killer)
            local nowClock = os.clock()

            if kSpeed > SPRINT_THRESHOLD then killerSprintLatch[killer] = nowClock end
            local latchAge    = killerSprintLatch[killer] and (nowClock - killerSprintLatch[killer]) or math.huge
            local isSprinting = kSpeed > SPRINT_THRESHOLD or latchAge < 0.06
            local effectiveRange = baseProximity + (isSprinting and RANGE_SPRINT_BONUS or 0)

            if detectionRangeSlider and effectiveRange ~= lastDisplayedRange then
                lastDisplayedRange = effectiveRange
                suppressCallback = true
                detectionRangeSlider:SetValue(effectiveRange)
                suppressCallback = false
            end

            updateRadiusPart(killer, khrp, effectiveRange, dist <= effectiveRange)

            if mode == "Auto Backstab" then
                local dot = (khrp.Position - hrp.Position).Unit:Dot(khrp.CFrame.LookVector)
                if dot > BACKSTAB_THRESHOLD_COS and dist <= effectiveRange and nowClock - lastTrigger >= COOLDOWN then
                    local cdNum = readCooldownValue(getDaggerCooldown())
                    if not (cdNum and cdNum > 0.1) then
                        lastTrigger = nowClock; bumpTriggerCount()
                        task.spawn(function()
                            aimAtKiller(killer, 0.5)
                            if daggerEnabled then tryActivateButton(getDaggerButton()) end
                            if AIM_SNAP_DELAY > 0 then
                                task.wait(AIM_SNAP_DELAY); if not isRunning then return end
                                local khrp2 = killer:FindFirstChild("HumanoidRootPart")
                                local char2 = clientPlayer.Character
                                local hrp2  = char2 and char2:FindFirstChild("HumanoidRootPart")
                                if khrp2 and hrp2 then
                                    hrp2.CFrame = CFrame.lookAt(hrp2.Position, Vector3.new(khrp2.Position.X, hrp2.Position.Y, khrp2.Position.Z), Vector3.new(0,1,0))
                                end
                            end
                        end)
                        break
                    end
                end

            elseif mode == "Dash-Stab Tech" then
                local dot     = (hrp.Position - khrp.Position).Unit:Dot(khrp.CFrame.LookVector)
                local inFront = dot > BACKSTAB_THRESHOLD_COS and dist <= effectiveRange
                local twoTimeCrouching = char:GetAttribute("Crouching") == true
                if inFront and twoTimeCrouching and nowClock - lastTrigger >= COOLDOWN then
                    local cdNum = readCooldownValue(getDaggerCooldown())
                    if not (cdNum and cdNum > 0.1) then
                        lastTrigger = nowClock; bumpTriggerCount()
                        task.spawn(function() performDashStabTech(killer) end)
                        break
                    end
                end
            end
        end
    end)
end)

local noclipEnabled    = false
local noclipConnection = nil

local function setNoclip(state)
    noclipEnabled = state
    if noclipEnabled then
        noclipConnection = runService.Stepped:Connect(function()
            local char = clientPlayer.Character
            if char then for _, p in ipairs(char:GetDescendants()) do if p:IsA("BasePart") then p.CanCollide = false end end end
        end)
    else
        if noclipConnection then noclipConnection:Disconnect(); noclipConnection = nil end
        local char = clientPlayer.Character
        if char then for _, p in ipairs(char:GetDescendants()) do if p:IsA("BasePart") then p.CanCollide = true end end end
    end
end

game:GetService("UserInputService").InputBegan:Connect(function(input, gpe)
    if gpe then return end
    if input.KeyCode == Enum.KeyCode.N then setNoclip(not noclipEnabled) end
end)
