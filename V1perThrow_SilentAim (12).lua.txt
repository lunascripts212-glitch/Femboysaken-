-- Utils
local motion = {}

local function getVel(hrp)
    local t = tick()
    local d = motion[hrp] or {p=hrp.Position, tm=t, v=Vector3.zero}
    motion[hrp] = d
    local dt = t - d.tm
    if dt > 0 then
        d.v = (hrp.Position - d.p) / dt
        d.p = hrp.Position
        d.tm = t
    end
    return d.v
end

local function nearestKiller(pos)
    local f = workspace.Players and workspace.Players.Killers
    if not f then return end
    local n, dst = nil, math.huge
    for _, m in f:GetChildren() do
        local h = m.HumanoidRootPart
        local hu = m:FindFirstChildOfClass("Humanoid")
        if h and hu and hu.Health > 0 then
            local d = (h.Position - pos).Magnitude
            if d < dst then dst, n = d, m end
        end
    end
    return n
end

-- Head-aim spoof CF
local function spoofCF(myHRP, kHRP, cfg)
    local v = getVel(kHRP)
    local v0, g = cfg.MaxSpeed, cfg.ProjectileArc
    local hip = (myHRP.Parent:FindFirstChildOfClass("Humanoid") or {}).HipHeight or 1.35
    local yOff = (hip + myHRP.Size.Y/2)/2
    local sp = myHRP.Position + Vector3.new(0,yOff,0)

    local pred = kHRP.Position + v * PREDICTION
    local tgt = pred + Vector3.new(0, HEAD_OFFSET, 0)

    local del = tgt - sp
    local flat = Vector3.new(del.X,0,del.Z)
    local dx, dy = flat.Magnitude, del.Y

    if dx < 0.1 then return CFrame.lookAt(Camera.CFrame.Position, Camera.CFrame.Position + (dy>=0 and Vector3.yAxis or -Vector3.yAxis)) end

    local vv = v0*v0
    local disc = vv*vv - g*(g*dx*dx + 2*dy*vv)
    local tanT = disc<0 and dy/dx or (vv - math.sqrt(disc))/(g*dx)

    local alpha = math.atan2(3*tanT-1, 3+tanT)

    local yaw = CFrame.lookAt(Camera.CFrame.Position, Camera.CFrame.Position + flat.Unit)
    return yaw * CFrame.Angles(alpha, 0, 0)
end

-- Patch
local patched, origCB = false, nil

local function actor()
    for _,a in Actors.CurrentActors do if a.Player == LocalPlayer then return a end end
end

local function applyPatch(act)
    if patched or not act or not act.Behavior?.Abilities?.Crystal then return end

    local RF = ReplicatedStorage.Modules.Network:FindFirstChild("RemoteFunction")

    origCB = act.Behavior.Abilities.Crystal.Callback

    act.Behavior.Abilities.Crystal.Callback = function(self, inp)
        if RunService:IsServer() then return origCB(self, inp) end

        local res = origCB(self, inp)

        if inp == 1 and _G.LT_Enabled then  -- FIRE input
            local hrp = self.Rig?.HumanoidRootPart
            if not hrp then return res end

            local kil = nearestKiller(hrp.Position)
            local khrp = kil?.HumanoidRootPart
            if not khrp then print("[LT] No killer") return res end

            print("[LT] Firing spoof - pred:", PREDICTION)

            local spoof = spoofCF(hrp, khrp, self.Config.Crystal)

            if isMobile and RF then
                local oldInvoke = getcallbackvalue and getcallbackvalue(RF, "OnClientInvoke") or RF.OnClientInvoke
                local oldDev = LocalPlayer:GetAttribute("Device") or "PC"

                LocalPlayer:SetAttribute("Device", "Mobile")
                task.wait(0.15)

                RF.OnClientInvoke = function(n, ...)
                    if n == "GetCameraCF" then
                        print("[LT] Mobile GetCameraCF spoofed")
                        RF.OnClientInvoke = oldInvoke
                        return spoof
                    end
                    return oldInvoke and oldInvoke(n, ...) 
                end

                task.delay(4, function()
                    if RF and (getcallbackvalue and getcallbackvalue(RF, "OnClientInvoke") or RF.OnClientInvoke) ~= oldInvoke then
                        RF.OnClientInvoke = oldInvoke
                    end
                    LocalPlayer:SetAttribute("Device", oldDev)
                end)
            else
                -- PC: brief scriptable + CFrame set (minimal visual, blocks input flick)
                local oldCam = Camera.CameraType
                Camera.CameraType = Enum.CameraType.Scriptable
                local oldCF = Camera.CFrame
                Camera.CFrame = CFrame.new(oldCF.Position, spoof.Position)  -- direction only
                RunService.Heartbeat:Wait()
                Camera.CFrame = oldCF  -- restore pos/rot if needed
                Camera.CameraType = oldCam
                print("[LT] PC camera nudge")
            end
        end
        return res
    end

    patched = true
    WindUI:Notify({Title="LoveThrow", Content="Patched - manual throw → head aim", Duration=4})
end

local function removePatch()
    local a = actor()
    if patched and a?.Behavior?.Abilities?.Crystal and origCB then
        a.Behavior.Abilities.Crystal.Callback = origCB
    end
    patched = false
end

-- Loop
_G.LT_Enabled = true
_G.LT_Unloaded = false

task.spawn(function()
    local prev = nil
    while not _G.LT_Unloaded do
        task.wait(0.6)
        local cur = actor()
        if cur ~= prev then
            motion = {}
            if prev then removePatch() WindUI:Notify({Title="Round change", Content="Repatching...", Duration=2}) end
            prev = cur
            if cur then applyPatch(cur) end
        end
    end
    removePatch()
end)

print("[LoveThrow] Ready - throw crystal manually, check console when firing")
